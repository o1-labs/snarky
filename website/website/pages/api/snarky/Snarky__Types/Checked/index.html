Module Snarky__Types.Checked

type ('a, 's, 'f) t =
| Pure : 'a -> ('a, 's, 'f) t
| Direct : (('s, 'f) Snarky.Run_state.t -> ('s, 'f) Snarky.Run_state.t * 'a) * ('a -> ('b, 's, 'f) t) -> ('b, 's, 'f) t
| Reduced : ('a, 's, 'f) t * (('s, 'f) Snarky.Run_state.t -> ('s, 'f) Snarky.Run_state.t) * 'a * ('a -> ('b, 's, 'f) t) -> ('b, 's, 'f) t
| Add_constraint : 'f Snarky.Cvar.t Snarky.Constraint.t * ('a, 's, 'f) t -> ('a, 's, 'f) t
| As_prover : (unit, 'f, 's) As_prover.t * ('a, 's, 'f) t -> ('a, 's, 'f) t
| With_label : string * ('a, 's, 'f) t * ('a -> ('b, 's, 'f) t) -> ('b, 's, 'f) t
| With_state : ('s1, 'f, 's) As_prover.t * ('s1 -> (unit, 'f, 's) As_prover.t) * ('b, 's1, 'f) t * ('b -> ('a, 's, 'f) t) -> ('a, 's, 'f) t
| With_handler : Snarky.Request.Handler.single * ('a, 's, 'f) t * ('a -> ('b, 's, 'f) t) -> ('b, 's, 'f) t
| Clear_handler : ('a, 's, 'f) t * ('a -> ('b, 's, 'f) t) -> ('b, 's, 'f) t
| Exists : ('var, 'value, 'f, (unit, unit, 'f) t) Typ.t * (('value Snarky.Request.t, 'f, 's) As_prover.t, ('value, 'f, 's) As_prover.t) Provider.t * (('var, 'value) Snarky.Handle.t -> ('a, 's, 'f) t) -> ('a, 's, 'f) t
| Next_auxiliary : (int -> ('a, 's, 'f) t) -> ('a, 's, 'f) t

The type ('ret, 'state, 'field, 'runner_state) t represents a checked computation, where

  • 'state is the type that holds the state used by As_prover computations
  • 'state -> 'ret is the type of the computation
  • 'field is the type of the field elements.
