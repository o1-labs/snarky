Module Snarky__Types.Typ

type ('var, 'value, 'field, 'r) t = {
store : 'value -> ('var, 'field) Snarky.Typ_monads.Store.t;
read : 'var -> ('value, 'field) Snarky.Typ_monads.Read.t;
alloc : ('var, 'field) Snarky.Typ_monads.Alloc.t;
check : 'var -> (unit, unit, 'field, 'r) Checked.t;
}

The type ('var, 'value, 'field, 'runner_state) t describes a mapping from OCaml types to the variables and constraints they represent:

  • 'value is the OCaml type
  • 'field is the type of the field elements
  • 'var is some other type that contains some R1CS variables
  • 'runner_state is the internal type of the checked computation evaluator.

For convenience and readability, it is usually best to have the 'var type mirror the 'value type in structure, for example:

type t = {b1 : bool; b2 : bool} (* 'value *)

let or (x : t) = x.b1 || x.b2

module Checked = struct
  type t = {b1 : Snark.Boolean.var; b2 : Snark.Boolean.var} (* 'var *)

  let or (x : t) = Snark.Boolean.(x.b1 || x.b2)
end
