type type_expr = {
mutable type_desc : type_desc;
type_id : int;
mutable type_depth : int;
}
and type_desc =
| Tvar of string option * Ast_types.explicitness
| Ttuple of type_expr list
| Tarrow of type_expr * type_expr * Ast_types.explicitness * Ast_types.arg_label
| Tctor of variant
| Tpoly of type_expr list * type_expr
and variant = {
var_ident : Ast_types.Longident.t;
var_params : type_expr list;
var_implicit_params : type_expr list;
var_decl : type_decl;
}
and field_decl = {
fld_ident : Ident.t;
fld_type : type_expr;
}
and ctor_args =
| Ctor_tuple of type_expr list
| Ctor_record of type_decl
and ctor_decl = {
ctor_ident : Ident.t;
ctor_args : ctor_args;
ctor_ret : type_expr option;
}
and type_decl = {
tdec_ident : Ident.t;
tdec_params : type_expr list;
tdec_implicit_params : type_expr list;
tdec_desc : type_decl_desc;
tdec_id : int;
}
and type_decl_desc =
| TAbstract
| TAlias of type_expr
| TUnfold of type_expr
| TRecord of field_decl list
| TVariant of ctor_decl list
| TOpen
| TExtend of Ast_types.Longident.t * type_decl * ctor_decl list

Internal; this should never be present in the AST.
